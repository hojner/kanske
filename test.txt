# Profiles for headless Sway testing
# start-sway.fish creates HEADLESS-1 at 1920x1080
# simulate-hotplug.fish adds two more then removes them

# Matches the initial single-monitor state
profile single {
    output HEADLESS-1 enable mode 1920x1080 position 0,0
}

# Matches after first hotplug event
profile double {
    output HEADLESS-1 enable mode 1920x1080 position 0,0
    output * enable mode 1920x1080 position 1920,0
}

# Matches after simulate-hotplug creates two additional outputs
profile triple {
    output HEADLESS-1 enable mode 1920x1080 position 0,0
    output * enable mode 1920x1080 position 1920,0
    output * enable mode 1920x1080 position 3840,0
}

output HEADLESS-1 enable mode 1920x1080 position 50,0
